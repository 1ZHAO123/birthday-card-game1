        // 主页面烟花粒子
        function createFireworks() {
            const colors = ['#ff4081', '#8000ff', '#00bfff', '#ffd700', '#32cd32'];
            const count = 15;
            const mainPage = document.getElementById('mainPage');
            
            function createParticle() {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                particle.style.left = `${Math.random() * 100}vw`;
                particle.style.animation = `float ${5 + Math.random() * 5}s linear infinite`;
                mainPage.appendChild(particle);
                
                setTimeout(() => {
                    mainPage.removeChild(particle);
                    createParticle();
                }, (5 + Math.random() * 5) * 1000);
            }
            
            for (let i = 0; i < count; i++) createParticle();
        }

        // 创建底部牌堆
        function createCardStack() {
            const cardStack = document.getElementById('cardStack');
            cardStack.innerHTML = '';
            const shuffled = [...cardContents].sort(() => Math.random() - 0.5);
            let activeCard = null;
            let timer = null;

            // 5行x4列布局
            for (let r = 0; r < 5; r++) {
                const row = document.createElement('div');
                row.className = 'card-row';
                row.style.zIndex = 5 - r;
                
                for (let c = 0; c < 4; c++) {
                    const index = r * 4 + c;
                    if (index >= shuffled.length) break; // 防止数组越界
                    const card = document.createElement('div');
                    card.className = 'card';
                    card.innerHTML = `
                        <div class="card-inner">
                            <div class="card-front">${shuffled[index]}</div>
                            <div class="card-back">祝福卡</div>
                        </div>
                    `;

                    // 抽卡逻辑：点击后卡牌边缘发光（与卡牌同大），然后移动到中间展示
                    card.addEventListener('click', () => {
                        // 重置上一张
                        if (activeCard) {
                            document.getElementById('cardDisplay').classList.remove('active');
                            clearTimeout(timer);
                        }

                        // 卡牌边缘发光（动画时长0.5s，仅边缘发光且范围与卡牌一致）
                        card.classList.add('selected');
                        setTimeout(() => card.classList.remove('selected'), 500);

                        // 克隆卡牌到中间展示区（保持原尺寸）
                        const displayCard = document.createElement('div');
                        displayCard.className = 'card flipped';
                        displayCard.innerHTML = card.innerHTML;
                        const cardDisplay = document.getElementById('cardDisplay');
                        cardDisplay.innerHTML = '';
                        cardDisplay.appendChild(displayCard);
                        
                        // 显示中间卡牌
                        cardDisplay.classList.add('active');

                        activeCard = displayCard;
                        // 5秒后收回
                        timer = setTimeout(() => {
                            cardDisplay.classList.remove('active');
                            activeCard = null;
                        }, 5000);
                    });

                    row.appendChild(card);
                }
                cardStack.appendChild(row);
            }
        }
